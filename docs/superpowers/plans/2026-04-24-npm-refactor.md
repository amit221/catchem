# NPM Package Refactor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Refactor CatchEm from a Claude Code plugin into a standalone npm package that installs globally, auto-sets up hooks/skills for detected platforms, and self-updates via a SessionStart hook.

**Architecture:** Single npm package installed globally via `npm install -g catchem`. The CLI provides `catchem setup` (writes hooks + skills into platform config dirs), `catchem tick` (passive catching, called by hooks), and `catchem collection` (TUI viewer). Setup writes a SessionStart hook that auto-updates the package daily. No plugin repo needed — the npm package IS the distribution.

**Tech Stack:** TypeScript (source), plain JS for tick script, ESM `.mjs` for TUI launcher, Node.js 20+

```typescript
import fs from "fs";
import path from "path";
import os from "os";

interface Platform {
  name: string;
  detected: boolean;
  install: (catchemBin: string, creaturesPath: string) => void;
}

function getCatchemBin(): string {
  // Resolve the globally installed catchem bin location
  return process.argv[1] || "catchem";
}

function getCatchemRoot(): string {
  // The npm package root (where creatures/ lives)
  return path.join(__dirname, "../..");
}

function detectClaudeCode(): Platform {
  const configDir = path.join(os.homedir(), ".claude");
  return {
    name: "Claude Code",
    detected: fs.existsSync(configDir),
    install: (catchemBin, creaturesPath) => installClaudeCode(configDir, catchemBin, creaturesPath),
  };
}

function installClaudeCode(configDir: string, catchemBin: string, creaturesPath: string): void {
  const settingsPath = path.join(configDir, "settings.json");

  // Read existing settings or create empty
  let settings: any = {};
  try {
    settings = JSON.parse(fs.readFileSync(settingsPath, "utf8"));
  } catch {}

  // Add hooks
  if (!settings.hooks) settings.hooks = {};

  const tickCommand = `node "${path.join(getCatchemRoot(), "scripts", "tick.js")}"`;
  const updateCommand = `node -e "const fs=require('fs'),p=require('path'),os=require('os');const f=p.join(os.homedir(),'.catchem','last-update-check');try{const t=Number(fs.readFileSync(f,'utf8'));if(Date.now()-t<86400000)process.exit(0)}catch{}try{require('child_process').execSync('npm update -g catchem',{stdio:'ignore'});fs.mkdirSync(p.dirname(f),{recursive:true});fs.writeFileSync(f,Date.now().toString())}catch{}"`;

  const hookEvents = ["UserPromptSubmit", "PostToolUse", "Stop"];
  for (const event of hookEvents) {
    if (!settings.hooks[event]) settings.hooks[event] = [];
    // Remove old catchem hooks
    settings.hooks[event] = settings.hooks[event].filter(
      (h: any) => !JSON.stringify(h).includes("catchem")
    );
    settings.hooks[event].push({
      type: "command",
      command: tickCommand,
    });
  }

  // SessionStart: tick + auto-update
  if (!settings.hooks.SessionStart) settings.hooks.SessionStart = [];
  settings.hooks.SessionStart = settings.hooks.SessionStart.filter(
    (h: any) => !JSON.stringify(h).includes("catchem")
  );
  settings.hooks.SessionStart.push({
    type: "command",
    command: tickCommand,
  });
  settings.hooks.SessionStart.push({
    type: "command",
    command: updateCommand,
  });

  fs.writeFileSync(settingsPath, JSON.stringify(settings, null, 2));

  // Write skills
  const skillsDir = path.join(configDir, "skills");
  fs.mkdirSync(skillsDir, { recursive: true });

  const collectionSkill = `---
name: catchem-collection
description: View your CatchEm creature collection, evolution stages, and discovery progress
---

Run the following command to open the CatchEm TUI collection viewer:

\`\`\`bash
node "${path.join(getCatchemRoot(), "scripts", "launch-collection.mjs")}"
\`\`\`

This launches an interactive terminal UI. Run it with the Bash tool.
`;

  fs.writeFileSync(path.join(skillsDir, "catchem-collection.md"), collectionSkill);
}

function getAllPlatforms(): Platform[] {
  return [
    detectClaudeCode(),
    // Future: detectCursor(), detectCopilot(), detectCodex(), detectOpenCode()
  ];
}

export function runSetup(auto: boolean = false): void {
  if (!auto) console.log("🎮 CatchEm Setup\n");

  const platforms = getAllPlatforms();
  const detected = platforms.filter((p) => p.detected);

  if (detected.length === 0) {
    if (!auto) {
      console.log("No supported platforms detected.");
      console.log("Supported: Claude Code, Cursor, GitHub Copilot, Codex, OpenCode");
    }
    return;
  }

  const catchemBin = getCatchemBin();
  const creaturesPath = path.join(getCatchemRoot(), "creatures", "creatures.json");

  if (!auto) {
    console.log("Detected platforms:");
    for (const p of detected) {
      console.log(`  ✓ ${p.name}`);
    }
    console.log();
  }

  for (const p of detected) {
    p.install(catchemBin, creaturesPath);
    if (!auto) console.log(`  ✅ ${p.name} — hooks & skills installed`);
  }

  if (!auto) console.log("\n🎉 Setup complete! Creatures will start appearing as you code.");
}
```

    collection();

  case "help":
  case undefined:
    printUsage();
    break;

import fs from "fs";
import path from "path";
import os from "os";
import { runSetup } from "../../src/cli/setup";

// Mock os.homedir to use a temp directory
const originalHomedir = os.homedir;

beforeEach(() => {

afterEach(() => {

describe("runSetup", () => {

  it("hooks reference tick.js", () => {
    runSetup(true);
    const settingsPath = path.join(tempHome, ".claude", "settings.json");
    const settings = JSON.parse(fs.readFileSync(settingsPath, "utf8"));
    const hookCmd = JSON.stringify(settings.hooks.UserPromptSubmit);
    expect(hookCmd).toContain("tick.js");
  });

  it("writes collection skill file", () => {
    runSetup(true);
    const skillPath = path.join(tempHome, ".claude", "skills", "catchem-collection.md");
    expect(fs.existsSync(skillPath)).toBe(true);
    const content = fs.readFileSync(skillPath, "utf8");
    expect(content).toContain("launch-collection.mjs");
  });

  it("includes auto-update hook in SessionStart", () => {
    runSetup(true);
    const settingsPath = path.join(tempHome, ".claude", "settings.json");
    const settings = JSON.parse(fs.readFileSync(settingsPath, "utf8"));
    const sessionHooks = JSON.stringify(settings.hooks.SessionStart);
    expect(sessionHooks).toContain("npm update -g catchem");
  });

  it("preserves existing settings", () => {
    const settingsPath = path.join(tempHome, ".claude", "settings.json");
    fs.writeFileSync(settingsPath, JSON.stringify({ existingKey: "keep" }, null, 2));
    runSetup(true);
    const settings = JSON.parse(fs.readFileSync(settingsPath, "utf8"));
    expect(settings.existingKey).toBe("keep");
    expect(settings.hooks).toBeDefined();

  it("does not duplicate hooks on re-run", () => {
    runSetup(true);
    runSetup(true);
    const settingsPath = path.join(tempHome, ".claude", "settings.json");
    const settings = JSON.parse(fs.readFileSync(settingsPath, "utf8"));
    // Should have exactly 1 tick hook per event, not 2
    expect(settings.hooks.UserPromptSubmit.length).toBe(1);
  });
});

- [ ] **Step 1: Update release workflow with npm publish step**

Replace `.github/workflows/release.yml` with:

```yaml
name: release
on:
  push:
    branches:
      - master
permissions:
  contents: write
  issues: write
  pull-requests: write
jobs:
  release-please:
    runs-on: ubuntu-latest
    steps:
      - uses: googleapis/release-please-action@v4
        id: release
        with:
          config-file: release-please-config.json
          manifest-file: .release-please-manifest.json
          target-branch: master

      - name: Auto-merge release PR
        if: ${{ steps.release.outputs.pr }}
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          PR_NUMBER=$(echo '${{ steps.release.outputs.pr }}' | jq -r '.number')
          gh pr merge "$PR_NUMBER" --repo "${{ github.repository }}" --squash

      - uses: googleapis/release-please-action@v4
        if: ${{ steps.release.outputs.pr }}
        id: release-after-merge
        with:
          config-file: release-please-config.json
          manifest-file: .release-please-manifest.json
          target-branch: master

      # Publish to npm when a release is created
      - uses: actions/checkout@v4
        if: ${{ steps.release.outputs.release_created || steps.release-after-merge.outputs.release_created }}

      - uses: actions/setup-node@v4
        if: ${{ steps.release.outputs.release_created || steps.release-after-merge.outputs.release_created }}
        with:
          node-version: '20'
          cache: 'npm'
          registry-url: 'https://registry.npmjs.org'

      - run: npm ci
        if: ${{ steps.release.outputs.release_created || steps.release-after-merge.outputs.release_created }}

      - run: npm publish
        if: ${{ steps.release.outputs.release_created || steps.release-after-merge.outputs.release_created }}
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}
```

3. `/catchem-collection` skill → opens TUI
4. SessionStart hook → auto-updates daily via `npm update -g catchem`
